\doxysubsection{Detailed Description}
\subsubsection*{template$<$typename real\+\_\+t$>$\newline
class G\+Matrix$<$ real\+\_\+t $>$}

Class to implement a memory efficient Givens\textquotesingle{} Rotation Matrix that can vanish the element $ a_{k+1, k} $ of a matrix $ A $. 

A Givens rotation matrix is an orthogonal such that $ G_{N \times N} = [g_{ij}]_{N \times N} $ where ~\newline
 \begin{equation} g_{ij} = \begin{array}{ll} 1 & i = j \ne k, k+1 \\ \cos{\theta} & i = j = k, k+1 \\ \sin{\theta} & i = k, j = k+1 \\ -\sin{\theta} & i = k+1, j = k \\ 0 & \text{otherwise} \end{array} \end{equation}

When the Givens Rotation Matrix is multiplied to another matrix $ B_{N \times N} = [b_{i,j}]_{N \times N} $ \begin{equation} C_{N \times N} = G_{N \times N} \cdot B_{N \times N} \end{equation} \begin{equation} \Rightarrow c_{ij} = \sum_{l=1}^N g_{i,l} b_{l,j} \end{equation} \begin{equation} \Rightarrow c_{ij} = \begin{array}{ll} g_{k,k} b_{k,j} + g_{k,k+1} b_{k+1,j} & i=k \\ g_{k+1,k} b_{k,j} + g_{k+1,k+1} b_{k+1,j} & i=k+1 \\ b_{i,j} & \text{otherwise} \end{array} \end{equation}

R matrix can be represented as $ R_{N \times N} = [r_{i,j}]_{N \times N} $ where \begin{equation} r_{i,j} = \begin{array}{lllll} 0 & \text{if} & j < i-1 & \text{or} & j > i+1 \end{array} \end{equation}

Thus, upon multiplying the R matrix with the rotation matrix, we get matrix $ C_{N \times N} $

\begin{equation} c_{ij} = \begin{array}{ll} g_{k,k} r_{k,j} + g_{k,k+1} r_{k+1,j} & i=k \\ g_{k+1,k} r_{k,j} + g_{k+1,k+1} r_{k+1,j} & i=k+1 \\ r_{i,j} & \text{otherwise} \end{array} \end{equation}

\begin{equation} \Rightarrow c_{ij} = \begin{array}{lll} g_{k,k} r_{k,k-1} & i=k & j=k-1 \\ g_{k,k} r_{k,k} + g_{k,k+1} r_{k+1,k} & i=k & j=k \\ g_{k,k} r_{k,k+1} + g_{k,k+1} r_{k+1,k+1} & i=k & j=k+1 \\ g_{k,k+1} r_{k+1,k+2} & i=k & j=k+2 \\ g_{k+1,k} r_{k,k-1} & i=k+1 & j=k-1 \\ g_{k+1,k} r_{k,k} + g_{k+1,k+1} r_{k+1,k} & i=k+1 & j=k \\ g_{k+1,k} r_{k,k+1} + g_{k+1,k+1} r_{k+1,k+1} & i=k+1 & j=k+1 \\ g_{k+1,k+1} r_{k+1,k+2} & i=k+1 & j=k+2 \\ r_{i,j} & \text{otherwise} \end{array} \end{equation}

But as we iterate over k and multiply the corresponding rotation marix, the sub-\/diagonal ( $ r_{i,i-1} $) becomes zero and the super-\/super-\/diagonal ( $ r_{i,i+2} $) fills up with non-\/zero values.

\begin{equation} \Rightarrow c_{ij} = \begin{array}{lll} 0 & i=k & j=k-1 \\ g_{k,k} r_{k,k} + g_{k,k+1} r_{k+1,k} & i=k & j=k \\ g_{k,k} r_{k,k+1} + g_{k,k+1} r_{k+1,k+1} & i=k & j=k+1 \\ g_{k,k+1} r_{k+1,k+2} & i=k & j=k+2 \\ 0 & i=k+1 & j=k-1 \\ 0 & i=k+1 & j=k \\ g_{k+1,k} r_{k,k+1} + g_{k+1,k+1} r_{k+1,k+1} & i=k+1 & j=k+1 \\ g_{k+1,k+1} r_{k+1,k+2} & i=k+1 & j=k+2 \\ r_{i,j} & \text{otherwise} \end{array} \end{equation}
