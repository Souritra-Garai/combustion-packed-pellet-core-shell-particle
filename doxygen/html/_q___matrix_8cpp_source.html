126 template<typename real_t>
127 void QMatrix<real_t>::multiply(
128  real_t *b,
129  real_t *x
130 ) {
131  // The calculation for x[i] is independent of other rows
132  // Thus parallelization is possitble
133  #pragma omp parallel for
134 
135  // For each row of the Q matrix except the last row
136  for (int i = 0; i < N-1; i++)
137  {
138  // The calculation performed at each column of Q
139  // is independent of other columns
140  // But x[i] is the cumulative sum of the multiplications
141  // across all the columns
142  #pragma omp parallel for reduction(+:x[i])
143 
144  // For each non zero element of Q in the row i
145  for (int j = 0; j <= i+1; j++)
146 
147  // Cumulative sum of products of the elements of Q and b
148  x[i] += getElement(i, j) * b[j];
149  }
150 
151  // Again, calculations performed at each column of Q
152  // is independent of other columns
153  // But x[N-1] is the cumulative sum of the multiplications
154  // across all the columns
155  #pragma omp parallel for reduction(+:x[N-1])
156 
157  // For the last row the iterations across
158  // columns go till N-1
159  for (int j = 0; j < N; j++)
160 
161  // Cumulative sum of products of the elements of Q and b
162  x[N-1] += getElement(N-1, j) * b[j];
163 }
