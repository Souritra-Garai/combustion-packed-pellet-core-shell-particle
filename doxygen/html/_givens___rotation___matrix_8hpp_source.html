23 /**
24  * @brief Class to implement a memory efficient
25  * Givens' Rotation Matrix that can vanish the element
26  * \f$ a_{k+1, k} \f$ of a tridiagonal matrix \f$ A \f$
27  *
28  * A Givens rotation matrix is an orthogonal such that
29  * \f$ G_{N \times N} = [g_{ij}]_{N \times N} \f$
30  * where
31  * \f{equation}{
32  * g_{ij} = \begin{array}{cc}
33  * 1 & i = j \ne k, k+1 \\
34  * \cos{\theta} & i = j = k, k+1 \\
35  * \sin{\theta} & i = k, j = k+1 \\
36  * -\sin{\theta} & i = k+1, j = k \\
37  * 0 & \text{otherwise}
38  * \end{array}
39  * \f}
40  *
41  * When the Givens Rotation Matrix is multiplied to another
42  * matrix \f$ B_{N \times N} = [b_{i,j}]_{N \times N} \f$
43  * \f{equation}{ C_{N \times N} = G_{N \times N} \cdot B_{N \times N} \f}
44  * \f{equation} \Rightarrow c_{ij} = \sum_{l=1}^N g_{i,l} b_{l,j} \f}
45  * \f{equation}{
46  * \Rightarrow c_{ij} = \begin{array}{cc}
47  * g_{k,k} b_{k,j} + g_{k,k+1} b_{k+1,j} & i=k \\
48  * g_{k+1,k} b_{k,j} + g_{k+1,k+1} b_{k+1,j} & i=k+1 \\
49  * b_{i,j} & \text{otherwise}
50  * \end{array}
51  * \f}
52  *
53  * @tparam real_t float, double or long double data types to represent real numbers
54  *
55  */
56 template<typename real_t>
57 class GivensRotationMatrix

75  /**
76  * @brief Multiplies the Givens Rotation Matrix to the
77  * Tridiagonal Matrix and updates the tridiagonal matrix in place
78  *
79  * @param matrix Tridiagonal Matrix that will be converted into
80  * upper tridiagonal matrix after multiplication
81  *
82  * A Tridiagonal matrix can be represented as \f$ T_{N \times N} = [t_{i,j}]_{N \times N} \f$
83  * where
84  * \f{eqnarray*}{
85  * t_{i,j} = 0, & \text{if} j < i-1 \text{or} j > i+1
86  * }
87  *
88  * Thus, upon multiplying the tridiagonal matrix with the rotation matrix,
89  * we get matrix \f$ C_{N \times N} \f$
90  *
91  * \f{equation}{
92  * c_{ij} = \begin{array}{cc}
93  * g_{k,k} t_{k,j} + g_{k,k+1} t_{k+1,j} & i=k \\
94  * g_{k+1,k} t_{k,j} + g_{k+1,k+1} t_{k+1,j} & i=k+1 \\
95  * t_{i,j} & \text{otherwise}
96  * \end{array}
97  *
98  * \f{equation}{
99  * \Rightarrow c_{ij} = \begin{array}{cc}
100  * g_{k,k} t_{k,k-1} & i=k & j=k-1 \\
101  * g_{k,k} t_{k,k} + g_{k,k+1} t_{k+1,k} & i=k & j=k \\
102  * g_{k,k} t_{k,k+1} + g_{k,k+1} t_{k+1,k+1} & i=k & j=k+1 \\
103  * g_{k,k+1} t_{k+1,k+2} & i=k & j=k+2 \\
104  *
105  * g_{k+1,k} t_{k,k-1} & i=k+1 & j=k-1 \\
106  * g_{k+1,k} t_{k,k} + g_{k+1,k+1} t_{k+1,k} & i=k+1 & j=k \\
107  * g_{k+1,k} t_{k,k+1} + g_{k+1,k+1} t_{k+1,k+1} & i=k+1 & j=k+1 \\
108  * g_{k+1,k+1} t_{k+1,k+2} & i=k+1 & j=k+2 \\
109  *
110  * t_{i,j} & \text{otherwise}
111  * \end{array}
112  *
113  * But as we iterate over k and multiply the corresponding rotation marix,
114  * the sub-diagonal (\f$ t_{i,i-1} \f$) becomes zero and the super-super-diagonal
115  * (\f$ t_{i,i+2} \f$) fills up with non-zero values.
116  *
117  * \f{equation}{
118  * \Rightarrow c_{ij} = \begin{array}{cc}
119  * 0 & i=k & j=k-1 \\
120  * g_{k,k} t_{k,k} + g_{k,k+1} t_{k+1,k} & i=k & j=k \\
121  * g_{k,k} t_{k,k+1} + g_{k,k+1} t_{k+1,k+1} & i=k & j=k+1 \\
122  * g_{k,k+1} t_{k+1,k+2} & i=k & j=k+2 \\
123  *
124  * 0 & i=k+1 & j=k-1 \\
125  * 0 & i=k+1 & j=k \\
126  * g_{k+1,k} t_{k,k+1} + g_{k+1,k+1} t_{k+1,k+1} & i=k+1 & j=k+1 \\
127  * g_{k+1,k+1} t_{k+1,k+2} & i=k+1 & j=k+2 \\
128  *
129  * t_{i,j} & \text{otherwise}
130  * \end{array}
131  */
132  void multiply(TridiagonalMatrix<real_t> &matrix);
133 
134  /**
135  * @brief Multiplies the Givens Rotation Matrix to the
136  * Lower Triangular Matrix and updates the Lower Triangular matrix in place
137  *
138  * @param matrix Lower Triangular Matrix passed as reference
139  *
140  * A Lower Triangular may be represented as matrix
141  * \f$ L_{N \times N} = [l_{i,j}]_{N \times N} \f$ where
142  * \f{eqnarray*}{
143  * l_{i,j} = 0, & j > i
144  * \f}
145  *
146  * Thus, when a rotation matrix is multiplied to a lower triangular matrix
147  * \f{equation}{ C_{N \times N} = G_{N \times N} \cdot L_{N \times N} \f}
148  * \f{equation} \Rightarrow c_{ij} = \sum_{l=1}^N g_{i,l} b_{l,j} \f}
149  * \f{equation}{
150  * \Rightarrow c_{ij} = \begin{array}{cc}
151  * g_{k,k} l_{k,j} + g_{k,k+1} l_{k+1,j} & i=k \\
152  * g_{k+1,k} l_{k,j} + g_{k+1,k+1} l_{k+1,j} & i=k+1 \\
153  * l_{i,j} & \text{otherwise}
154  * \end{array}
155  * \f}
156  */
157  void multiply(LowerTriangularMatrix<real_t> &matrix);
