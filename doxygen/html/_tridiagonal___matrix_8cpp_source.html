107 template<typename real_t>
108 unsigned int TridiagonalMatrix<real_t>::getIndex(
109  unsigned int i,
110  unsigned int j
111 )
112 {
113  // In the current implementation of 2D Tridiagonal Matrix as
114  // a flattened array, the elements of i th row of the 2D Tridiagonal matrix
115  // is present in the 4 consecutive elements starting at index # 4*i
116  // 4*i is the element in the sub diagonal in the i th row \f$\Rightarrow (i,i-1)\f$
117  // 4*i + 1 is the element in the main diagonal in the i th row \f$\Rightarrow (i,i)\f$
118  // 4*i + 2 is the element in the super diagonal in the i th row \f$\Rightarrow (i,i+1)\f$
119  // 4*i + 3 is the element in the super super diagonal in the i th row \f$\Rightarrow (i,i+2)\f$
120  return 3*i + (j + 1);
121 }
