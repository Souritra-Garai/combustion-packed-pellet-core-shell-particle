23 /**
24  * @brief Class to implement a memory efficient
25  * Givens' Rotation Matrix that can vanish the element
26  * \f$ a_{k+1, k} \f$ of a tridiagonal matrix \f$ A \f$
27  *
28  * A Givens rotation matrix is an orthogonal such that
29  * \f$ G_{N \times N} = [g_{ij}]_{N \times N} \f$
30  * where
31  * \f{equation}{
32  * g_{ij} = \begin{array}{cc}
33  * 1 & i = j \ne k, k+1 \\
34  * \cos{\theta} & i = j = k, k+1 \\
35  * \sin{\theta} & i = k, j = k+1 \\
36  * -\sin{\theta} & i = k+1, j = k \\
37  * 0 & \textbf{otherwise}
38  * \end{array}
39  * \f}
40  *
41  * When the Givens Rotation Matrix is multiplied to another
42  * matrix \f$ B_{N \times N} = [b_{i,j}]_{N \times N} \f$
43  * \f{equation}{ C_{N \times N} = G_{N \times N} \cdot B_{N \times N} \f}
44  * \f{equation} \Rightarrow c_{ij} = \sum_{l=1}^N g_{i,l} b_{l,j} \f}
45  * \f{equation}{
46  * \Rightarrow c_{ij} = \begin{array}{cc}
47  * g_{k,k} b_{k,j} + g_{k,k+1} b_{k+1,j} & i=k \\
48  * g_{k+1,k} b_{k+1,j} + g_{k+1,k+1} b_{k+1,j} & i=k+1 \\
49  * b_{i,j} & \textbf{otherwise}
50  * \end{array}
51  * \f}
52  *
53  * @tparam real_t float, double or long double data types to represent real numbers
54  *
55  */
