25 /**
26  * @brief Class to implement a memory efficient
27  * Givens' Rotation Matrix that can vanish the element
28  * \f$ a_{k+1, k} \f$ of a R matrix \f$ A \f$
29  *
30  * A Givens rotation matrix is an orthogonal such that
31  * \f$ G_{N \times N} = [g_{ij}]_{N \times N} \f$
32  * where
33  * \f{equation}{
34  * g_{ij} = \begin{array}{cc}
35  * 1 & i = j \ne k, k+1 \\
36  * \cos{\theta} & i = j = k, k+1 \\
37  * \sin{\theta} & i = k, j = k+1 \\
38  * -\sin{\theta} & i = k+1, j = k \\
39  * 0 & \text{otherwise}
40  * \end{array}
41  * \f}
42  *
43  * When the Givens Rotation Matrix is multiplied to another
44  * matrix \f$ B_{N \times N} = [b_{i,j}]_{N \times N} \f$
45  * \f{equation}{ C_{N \times N} = G_{N \times N} \cdot B_{N \times N} \f}
46  * \f{equation} \Rightarrow c_{ij} = \sum_{l=1}^N g_{i,l} b_{l,j} \f}
47  * \f{equation}{
48  * \Rightarrow c_{ij} = \begin{array}{cc}
49  * g_{k,k} b_{k,j} + g_{k,k+1} b_{k+1,j} & i=k \\
50  * g_{k+1,k} b_{k,j} + g_{k+1,k+1} b_{k+1,j} & i=k+1 \\
51  * b_{i,j} & \text{otherwise}
52  * \end{array}
53  * \f}
54  *
55  * @tparam real_t float, double or long double data types to represent real numbers
56  */
57 template<typename real_t>
58 class GMatrix

76  /**
77  * @brief Multiplies the Givens Rotation Matrix to the
78  * R Matrix and updates the R matrix in place
79  *
80  * @param R R Matrix that will be converted into
81  * upper tridiagonal matrix after multiplication
82  *
83  * R matrix can be represented as \f$ R_{N \times N} = [r_{i,j}]_{N \times N} \f$
84  * where
85  * \f{eqnarray*}{
86  * r_{i,j} = 0, & \text{if} j < i-1 \text{or} j > i+1
87  * \f}
88  *
89  * Thus, upon multiplying the R matrix with the rotation matrix,
90  * we get matrix \f$ C_{N \times N} \f$
91  *
92  * \f{equation}{
93  * c_{ij} = \begin{array}{cc}
94  * g_{k,k} r_{k,j} + g_{k,k+1} r_{k+1,j} & i=k \\
95  * g_{k+1,k} r_{k,j} + g_{k+1,k+1} r_{k+1,j} & i=k+1 \\
96  * r_{i,j} & \text{otherwise}
97  * \end{array}
98  * \f}
99  *
100  * \f{equation}{
101  * \Rightarrow c_{ij} = \begin{array}{ccc}
102  * g_{k,k} r_{k,k-1} & i=k & j=k-1 \\
103  * g_{k,k} r_{k,k} + g_{k,k+1} r_{k+1,k} & i=k & j=k \\
104  * g_{k,k} r_{k,k+1} + g_{k,k+1} r_{k+1,k+1} & i=k & j=k+1 \\
105  * g_{k,k+1} r_{k+1,k+2} & i=k & j=k+2 \\
106  *
107  * g_{k+1,k} r_{k,k-1} & i=k+1 & j=k-1 \\
108  * g_{k+1,k} r_{k,k} + g_{k+1,k+1} r_{k+1,k} & i=k+1 & j=k \\
109  * g_{k+1,k} r_{k,k+1} + g_{k+1,k+1} r_{k+1,k+1} & i=k+1 & j=k+1 \\
110  * g_{k+1,k+1} r_{k+1,k+2} & i=k+1 & j=k+2 \\
111  *
112  * r_{i,j} & \text{otherwise}
113  * \end{array}
114  * \f}
115  *
116  * But as we iterate over k and multiply the corresponding rotation marix,
117  * the sub-diagonal (\f$ r_{i,i-1} \f$) becomes zero and the super-super-diagonal
118  * (\f$ r_{i,i+2} \f$) fills up with non-zero values.
119  *
120  * \f{equation}{
121  * \Rightarrow c_{ij} = \begin{array}{ccc}
122  * 0 & i=k & j=k-1 \\
123  * g_{k,k} r_{k,k} + g_{k,k+1} r_{k+1,k} & i=k & j=k \\
124  * g_{k,k} r_{k,k+1} + g_{k,k+1} r_{k+1,k+1} & i=k & j=k+1 \\
125  * g_{k,k+1} r_{k+1,k+2} & i=k & j=k+2 \\
126  *
127  * 0 & i=k+1 & j=k-1 \\
128  * 0 & i=k+1 & j=k \\
129  * g_{k+1,k} r_{k,k+1} + g_{k+1,k+1} r_{k+1,k+1} & i=k+1 & j=k+1 \\
130  * g_{k+1,k+1} r_{k+1,k+2} & i=k+1 & j=k+2 \\
131  *
132  * r_{i,j} & \text{otherwise}
133  * \end{array}
134  * \f}
135  */
136  void multiply(RMatrix<real_t> &R);

143  *
144  * \f{equation}{
145  * \Rightarrow c_{ij} = \begin{array}{ccc}
146  * 0 & i=k & j=k-1 \\
147  * g_{N-1,N-1} r_{N-1,N-1} + g_{N-1,N} r_{N,N-1} & i=k & j=k \\
148  * g_{N-1,N-1} r_{N-1,N} + g_{N-1,N} r_{N,N} & i=k & j=k+1 \\
149  *
150  * 0 & i=k+1 & j=k-1 \\
151  * 0 & i=k+1 & j=k \\
152  * g_{N,N-1} r_{N-1,N} + g_{N,N} r_{N,N} & i=k+1 & j=k+1 \\

154  * r_{i,j} & \text{otherwise}

181  * g_{k+1,k} l_{k,j} + g_{k+1,k+1} q_{k+1,j} & i=k+1 \\
182  * q_{i,j} & \text{otherwise}
183  * \end{array}
184  * \f}
185  */
